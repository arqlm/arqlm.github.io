    "import numpy as np\n",
    "import bokeh.plotting\n",
    "import bokeh.io\n",
    "\n",
    "bokeh.io.output_notebook()\n",
    "\n",
    "# Generar valores para graficar\n",
    "t = np.linspace(0, 2*np.pi, 200)\n",
    "x = 16 * np.sin(t)**3\n",
    "y = 13 * np.cos(t) - 5 * np.cos(2*t) - 2 * np.cos(3*t) - np.cos(4*t)\n",
    "\n",
    "p = bokeh.plotting.figure(height=250, width=275)\n",
    "p.line(x, y, color='red', line_width=3)\n",
    "text = bokeh.models.Label(x=0, y=0, text='BE/Bi 103 a', text_align='center')\n",
    "p.add_layout(text)\n",
    "\n",
    "bokeh.io.show(p)"
